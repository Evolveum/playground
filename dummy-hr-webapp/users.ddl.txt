CREATE TABLE usershr

(
  
	id serial NOT NULL,
	firstname character(255),

 	lastname character(255),

  	emailaddress character(255),

  	employeenumber integer,

  	artname character(255),

  	emptype emp NOT NULL

  	status emp NOT NULL

)
